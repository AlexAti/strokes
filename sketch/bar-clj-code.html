(def data [63 39 31 53 25 32 175 69 51])
(def x (-> d3 .-scale (.linear…)))   ; data ↦ width
(def y (-> d3 .-scale (.ordinal…)))  ; index ↦ y

; Data ↦ Element
(def rect (-> svg (.selectAll "rect"))
    (.data data))

; Data Attributes ↦ Element Attributes
(-> rect (.enter) (.append "rect")
    (.attr {:width  identity
            :y      #(identity %2)
            :height (.rangeBand y)}))
